class ListNode:
    def __init__(self, val=0):
        self.val = val
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None
    
    # O(1) - –≤—Å—Ç–∞–≤–∫–∞ –≤ –Ω–∞—á–∞–ª–æ
    def prepend(self, val):
        new_node = ListNode(val)
        new_node.next = self.head
        self.head = new_node
    
    # O(n) - –ø–æ–∏—Å–∫
    def find(self, val):
        current = self.head
        while current:
            if current.val == val:
                return current
            current = current.next
        return None
    
    # O(n) - —É–¥–∞–ª–µ–Ω–∏–µ
    def delete(self, val):
        if not self.head:
            return
        
        if self.head.val == val:
            self.head = self.head.next
            return
        
        current = self.head
        while current.next:
            if current.next.val == val:
                current.next = current.next.next
                return
            current = current.next

class Stack:
    def __init__(self):
        self.items = []
    
    # O(1)
    def push(self, item):
        self.items.append(item)
    
    # O(1)
    def pop(self):
        if self.is_empty():
            raise IndexError("Pop from empty stack")
        return self.items.pop()
    
    # O(1)
    def peek(self):
        if self.is_empty():
            return None
        return self.items[-1]
    
    def is_empty(self):
        return len(self.items) == 0

# –ü—Ä–∏–º–µ—Ä: –ø—Ä–æ–≤–µ—Ä–∫–∞ —Å–∫–æ–±–æ–∫
def is_valid_parentheses(s):
    stack = Stack()
    pairs = {'(': ')', '[': ']', '{': '}'}
    
    for char in s:
        if char in pairs:  # –æ—Ç–∫—Ä—ã–≤–∞—é—â–∞—è —Å–∫–æ–±–∫–∞
            stack.push(char)
        elif char in pairs.values():  # –∑–∞–∫—Ä—ã–≤–∞—é—â–∞—è
            if stack.is_empty():
                return False
            if pairs[stack.pop()] != char:
                return False
    
    return stack.is_empty()

# "(())" ‚Üí True, "([)]" ‚Üí False

from collections import deque

class Queue:
    def __init__(self):
        self.items = deque()
    
    # O(1)
    def enqueue(self, item):
        self.items.append(item)
    
    # O(1)
    def dequeue(self):
        if self.is_empty():
            raise IndexError("Dequeue from empty queue")
        return self.items.popleft()
    
    def is_empty(self):
        return len(self.items) == 0

# –ü—Ä–∏–º–µ—Ä: BFS –æ–±—Ö–æ–¥
def bfs_tree(root):
    if not root:
        return []
    
    result = []
    queue = Queue()
    queue.enqueue(root)
    
    while not queue.is_empty():
        node = queue.dequeue()
        result.append(node.val)
        
        if node.left:
            queue.enqueue(node.left)
        if node.right:
            queue.enqueue(node.right)
    
    return result
```

class HashTable:
    def __init__(self, size=10):
        self.size = size
        self.table = [[] for _ in range(size)]
    
    def _hash(self, key):
        return hash(key) % self.size
    
    # O(1) —Å—Ä–µ–¥–Ω–∏–π —Å–ª—É—á–∞–π
    def put(self, key, value):
        index = self._hash(key)
        bucket = self.table[index]
        
        # –û–±–Ω–æ–≤–ª—è–µ–º —Å—É—â–µ—Å—Ç–≤—É—é—â–∏–π –∫–ª—é—á
        for i, (k, v) in enumerate(bucket):
            if k == key:
                bucket[i] = (key, value)
                return
        
        # –î–æ–±–∞–≤–ª—è–µ–º –Ω–æ–≤—ã–π
        bucket.append((key, value))
    
    # O(1) —Å—Ä–µ–¥–Ω–∏–π —Å–ª—É—á–∞–π
    def get(self, key):
        index = self._hash(key)
        bucket = self.table[index]
        
        for k, v in bucket:
            if k == key:
                return v
        
        raise KeyError(key)
    
    def delete(self, key):
        index = self._hash(key)
        bucket = self.table[index]
        
        for i, (k, v) in enumerate(bucket):
            if k == key:
                del bucket[i]
                return
        
        raise KeyError(key)

class Node:
    def __init__(self, key=0, val=0):
        self.key = key
        self.val = val
        self.prev = None
        self.next = None

class LRUCache:
    def __init__(self, capacity):
        self.capacity = capacity
        self.cache = {}  # key -> node
        
        # –°–æ–∑–¥–∞–µ–º dummy –≥–æ–ª–æ–≤—ã –∏ —Ö–≤–æ—Å—Ç—ã
        self.head = Node()
        self.tail = Node()
        self.head.next = self.tail
        self.tail.prev = self.head
    
    def _add_node(self, node):
        """–î–æ–±–∞–≤–∏—Ç—å —É–∑–µ–ª –ø–æ—Å–ª–µ –≥–æ–ª–æ–≤—ã"""
        node.prev = self.head
        node.next = self.head.next
        
        self.head.next.prev = node
        self.head.next = node
    
    def _remove_node(self, node):
        """–£–¥–∞–ª–∏—Ç—å —É–∑–µ–ª"""
        prev_node = node.prev
        new_node = node.next
        
        prev_node.next = new_node
        new_node.prev = prev_node
    
    def _move_to_head(self, node):
        """–ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å –≤ –Ω–∞—á–∞–ª–æ (most recent)"""
        self._remove_node(node)
        self._add_node(node)
    
    def _pop_tail(self):
        """–£–¥–∞–ª–∏—Ç—å –ø–æ—Å–ª–µ–¥–Ω–∏–π —É–∑–µ–ª"""
        last_node = self.tail.prev
        self._remove_node(last_node)
        return last_node
    
    def get(self, key):
        node = self.cache.get(key)
        
        if node:
            # –ü–µ—Ä–µ–º–µ—â–∞–µ–º –≤ –Ω–∞—á–∞–ª–æ
            self._move_to_head(node)
            return node.val
        
        return -1
    
    def put(self, key, value):
        node = self.cache.get(key)
        
        if node:
            # –û–±–Ω–æ–≤–ª—è–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ
            node.val = value
            self._move_to_head(node)
        else:
            new_node = Node(key, value)
            
            if len(self.cache) >= self.capacity:
                # –£–¥–∞–ª—è–µ–º LRU
                tail = self._pop_tail()
                del self.cache[tail.key]
            
            self.cache[key] = new_node
            self._add_node(new_node)

cache.put(3, 3)      # evicts key 2
print(cache.get(2))  # -1 (not found)
```

class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

class BST:
    def __init__(self):
        self.root = None
    
    def insert(self, val):
        self.root = self._insert(self.root, val)
    
    def _insert(self, node, val):
        if not node:
            return TreeNode(val)
        
        if val < node.val:
            node.left = self._insert(node.left, val)
        else:
            node.right = self._insert(node.right, val)
        
        return node
    
    def search(self, val):
        return self._search(self.root, val)
    
    def _search(self, node, val):
        if not node or node.val == val:
            return node
        
        if val < node.val:
            return self._search(node.left, val)
        else:
            return self._search(node.right, val)
    
    def inorder(self):
        """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫"""
        result = []
        self._inorder(self.root, result)
        return result
    
    def _inorder(self, node, result):
        if node:
            self._inorder(node.left, result)
            result.append(node.val)
            self._inorder(node.right, result)

import heapq

class MinHeap:
    def __init__(self):
        self.heap = []
    
    def push(self, val):
        heapq.heappush(self.heap, val)
    
    def pop(self):
        if not self.heap:
            raise IndexError("pop from empty heap")
        return heapq.heappop(self.heap)
    
    def peek(self):
        if not self.heap:
            return None
        return self.heap[0]
    
    def size(self):
        return len(self.heap)

# Max Heap (–∏–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ–º –∑–Ω–∞—á–µ–Ω–∏—è)
class MaxHeap:
    def __init__(self):
        self.heap = []
    
    def push(self, val):
        heapq.heappush(self.heap, -val)
    
    def pop(self):
        if not self.heap:
            raise IndexError("pop from empty heap")
        return -heapq.heappop(self.heap)
    
    def peek(self):
        if not self.heap:
            return None
        return -self.heap[0]

# –ü—Ä–∏–º–µ—Ä: K largest elements
def k_largest(nums, k):
    min_heap = MinHeap()
    
    for num in nums:
        min_heap.push(num)
        if min_heap.size() > k:
            min_heap.pop()
    
    return [min_heap.pop() for _ in range(min_heap.size())]

print(k_largest([3,2,1,5,6,4], 2))  # [5, 6]

from collections import deque

def dfs_recursive(graph, start, visited=None):
    if visited is None:
        visited = set()
    
    visited.add(start)
    print(start, end=' ')
    
    for neighbor in graph.get_neighbors(start):
        if neighbor not in visited:
            dfs_recursive(graph, neighbor, visited)

def dfs_iterative(graph, start):
    visited = set()
    stack = [start]
    
    while stack:
        vertex = stack.pop()
        if vertex not in visited:
            visited.add(vertex)
            print(vertex, end=' ')
            
            # –î–æ–±–∞–≤–ª—è–µ–º —Å–æ—Å–µ–¥–µ–π –≤ —Å—Ç–µ–∫
            for neighbor in graph.get_neighbors(vertex):

                    stack.append(neighbor)

def bfs(graph, start):
    visited = set()
    queue = deque([start])
    visited.add(start)
    
    while queue:
        vertex = queue.popleft()
        print(vertex, end=' ')
        
        for neighbor in graph.get_neighbors(vertex):
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

# –ü—Ä–∏–º–µ—Ä: –ø–æ–∏—Å–∫ –ø—É—Ç–∏
def find_path_bfs(graph, start, end):
    if start == end:
        return [start]
    
    visited = {start}
    queue = deque([(start, [start])])
    
    while queue:
        vertex, path = queue.popleft()
        
        for neighbor in graph.get_neighbors(vertex):
            if neighbor not in visited:
                new_path = path + [neighbor]
                
                if neighbor == end:
                    return new_path
                
                visited.add(neighbor)
                queue.append((neighbor, new_path))
    
    return None  # –ø—É—Ç—å –Ω–µ –Ω–∞–π–¥–µ–Ω

class TrieNode:

class Trie:
    def __init__(self):
        self.root = TrieNode()
    
    def insert(self, word):
        node = self.root
        
        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
        
        node.is_end_word = True
    
    def search(self, word):
        node = self.root
        
        for char in word:
            if char not in node.children:
                return False
            node = node.children[char]
        
        return node.is_end_word

class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))  # –∫–∞–∂–¥—ã–π —ç–ª–µ–º–µ–Ω—Ç - —Å–≤–æ–π —Ä–æ–¥–∏—Ç–µ–ª—å
        self.rank = [0] * n           # –≤—ã—Å–æ—Ç–∞ –¥–µ—Ä–µ–≤–∞
        self.components = n           # –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç
    
    def find(self, x):
        """–ù–∞–π—Ç–∏ –∫–æ—Ä–µ–Ω—å —Å path compression"""
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])  # —Å–∂–∞—Ç–∏–µ –ø—É—Ç–∏
        return self.parent[x]
    
    def union(self, x, y):
        """–û–±—ä–µ–¥–∏–Ω–∏—Ç—å –º–Ω–æ–∂–µ—Å—Ç–≤–∞ x –∏ y"""
        root_x = self.find(x)
        root_y = self.find(y)
        
        if root_x == root_y:
            return False  # —É–∂–µ –≤ –æ–¥–Ω–æ–º –º–Ω–æ–∂–µ—Å—Ç–≤–µ
        
        # Union by rank - –ø—Ä–∏—Å–æ–µ–¥–∏–Ω—è–µ–º –º–µ–Ω—å—à–µ–µ –∫ –±–æ–ª—å—à–µ–º—É
        if self.rank[root_x] < self.rank[root_y]:
            self.parent[root_x] = root_y
        elif self.rank[root_x] > self.rank[root_y]:
            self.parent[root_y] = root_x
        else:
            self.parent[root_y] = root_x
            self.rank[root_x] += 1
        
        self.components -= 1
        return True
    
    def connected(self, x, y):
        """–ü—Ä–æ–≤–µ—Ä–∏—Ç—å, —Å–≤—è–∑–∞–Ω—ã –ª–∏ —ç–ª–µ–º–µ–Ω—Ç—ã"""
        return self.find(x) == self.find(y)
    
    def count_components(self):
        return self.components

def num_islands(grid):
    if not grid or not grid[0]:
        return 0
    
    rows, cols = len(grid), len(grid[0])
    uf = UnionFind(rows * cols)
    
    # –ü–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ–º –∏–∑–Ω–∞—á–∞–ª—å–Ω–æ —Ç–æ–ª—å–∫–æ –≤–æ–¥—É
    water_cells = 0
    
    for i in range(rows):
        for j in range(cols):
            if grid[i][j] == '0':  # –≤–æ–¥–∞
                water_cells += 1
            else:  # –∑–µ–º–ª—è, –ø—Ä–æ–≤–µ—Ä—è–µ–º —Å–æ—Å–µ–¥–µ–π
                # –°–æ–µ–¥–∏–Ω—è–µ–º —Å —Å–æ—Å–µ–¥–Ω–∏–º–∏ –∑–µ–º–ª—è–º–∏
                for di, dj in [(-1,0), (1,0), (0,-1), (0,1)]:
                    ni, nj = i + di, j + dj
                    if (0 <= ni < rows and 0 <= nj < cols and 
                        grid[ni][nj] == '1'):
                        uf.union(i * cols + j, ni * cols + nj)
    
    # –û–±—â–µ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç –º–∏–Ω—É—Å –≤–æ–¥–Ω—ã–µ –∫–ª–µ—Ç–∫–∏
    return uf.count_components() - water_cells

# –ü—Ä–∏–º–µ—Ä: –ø—Ä–æ–≤–µ—Ä–∫–∞ —Å–≤—è–∑–Ω–æ—Å—Ç–∏ –≥—Ä–∞—Ñ–∞
def is_connected_graph(n, edges):
    uf = UnionFind(n)
    
    for u, v in edges:
        uf.union(u, v)
    
    return uf.count_components() == 1

from collections import deque
import time

class RateLimiter:
    def __init__(self, max_requests=100, window_seconds=3600):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.requests = {}  # user_id -> deque of timestamps
    
    def is_allowed(self, user_id):
        now = time.time()
        
        if user_id not in self.requests:
            self.requests[user_id] = deque()
        
        user_requests = self.requests[user_id]
        
        # –£–¥–∞–ª—è–µ–º —Å—Ç–∞—Ä—ã–µ –∑–∞–ø—Ä–æ—Å—ã

            return True
        
        return False

class NotificationSystem:

        self.followers = defaultdict(set)  # user_id -> set of followers

        self.trending = []                 # min heap –¥–ª—è trending topics
        self.max_feed_size = 1000

        notification = {
            'user_id': user_id,
            'message': message,
            'timestamp': timestamp,
            'priority': priority
        }
        
        # –î–æ–±–∞–≤–ª—è–µ–º –≤ –ª–µ–Ω—Ç—ã –≤—Å–µ—Ö –ø–æ–¥–ø–∏—Å—á–∏–∫–æ–≤

        feed.append(notification)
        
        # –û–≥—Ä–∞–Ω–∏—á–∏–≤–∞–µ–º —Ä–∞–∑–º–µ—Ä –ª–µ–Ω—Ç—ã
        if len(feed) > self.max_feed_size:
            feed.popleft()
    
    def get_feed(self, user_id, limit=10):
        feed = self.feeds[user_id]
        # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –ø–æ—Å–ª–µ–¥–Ω–∏–µ N —É–≤–µ–¥–æ–º–ª–µ–Ω–∏–π
        return list(feed)[-limit:]
    
    def get_trending(self, limit=5):
        # –¢–æ–ø trending —É–≤–µ–¥–æ–º–ª–µ–Ω–∏–π
        trending = heapq.nsmallest(limit, self.trending)

class SearchEngine:
    def __init__(self):
        self.documents = {}        # doc_id -> content
        self.word_index = defaultdict(set)  # word -> set of doc_ids
        self.trie = Trie()        # –î–ª—è –∞–≤—Ç–æ–¥–æ–ø–æ–ª–Ω–µ–Ω–∏—è
        self.doc_scores = {}      # doc_id -> relevance score
    
    def add_document(self, doc_id, content):
        self.documents[doc_id] = content
        words = content.lower().split()
        
        # –ò–Ω–¥–µ–∫—Å–∏—Ä—É–µ–º —Å–ª–æ–≤–∞
        for word in set(words):  # –£–Ω–∏–∫–∞–ª—å–Ω—ã–µ —Å–ª–æ–≤–∞
            self.word_index[word].add(doc_id)
            self.trie.insert(word)
        
        # –ü—Ä–æ—Å—Ç–æ–π scoring: –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —É–Ω–∏–∫–∞–ª—å–Ω—ã—Ö —Å–ª–æ–≤
        self.doc_scores[doc_id] = len(set(words))
    
    def search(self, query):
        query_words = query.lower().split()
        
        if not query_words:
            return []
        
        # –ù–∞—Ö–æ–¥–∏–º –¥–æ–∫—É–º–µ–Ω—Ç—ã, —Å–æ–¥–µ—Ä–∂–∞—â–∏–µ –≤—Å–µ —Å–ª–æ–≤–∞
        result_docs = self.word_index[query_words[0]].copy()
        
        for word in query_words[1:]:
            result_docs &= self.word_index[word]  # –ü–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤
        
        # –°–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ —Ä–µ–ª–µ–≤–∞–Ω—Ç–Ω–æ—Å—Ç–∏
        scored_docs = [(self.doc_scores[doc_id], doc_id) 
                      for doc_id in result_docs]
        scored_docs.sort(reverse=True)
        
        return [doc_id for _, doc_id in scored_docs]
    
    def autocomplete(self, prefix):
        return self.trie.get_words_with_prefix(prefix)[:5]
    
    def get_document(self, doc_id):
        return self.documents.get(doc_id)
