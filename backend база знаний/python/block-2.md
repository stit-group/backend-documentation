        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –ø—Ä–∞–≤–∏–ª–∞ —Ç—Ä–µ—É–≥–æ–ª—å–Ω–∏–∫–∞"""
        return (self.side_a + self.side_b > self.side_c and
                self.side_a + self.side_c > self.side_b and
                self.side_b + self.side_c > self.side_a)
    
    def area(self):
        """–ü–ª–æ—â–∞–¥—å —Ç—Ä–µ—É–≥–æ–ª—å–Ω–∏–∫–∞ –ø–æ —Ñ–æ—Ä–º—É–ª–µ –ì–µ—Ä–æ–Ω–∞"""
        s = self.perimeter() / 2  # –ü–æ–ª—É–ø–µ—Ä–∏–º–µ—Ç—Ä
        return math.sqrt(s * (s - self.side_a) * (s - self.side_b) * (s - self.side_c))
    
    def perimeter(self):
        """–ü–µ—Ä–∏–º–µ—Ç—Ä —Ç—Ä–µ—É–≥–æ–ª—å–Ω–∏–∫–∞"""
        return self.side_a + self.side_b + self.side_c
    
    def triangle_type(self):
        """–û–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —Ç–∏–ø–∞ —Ç—Ä–µ—É–≥–æ–ª—å–Ω–∏–∫–∞"""
        sides = sorted([self.side_a, self.side_b, self.side_c])
        
        if sides[0] == sides[1] == sides[2]:
            return "—Ä–∞–≤–Ω–æ—Å—Ç–æ—Ä–æ–Ω–Ω–∏–π"
        elif sides[0] == sides[1] or sides[1] == sides[2]:
            return "—Ä–∞–≤–Ω–æ–±–µ–¥—Ä–µ–Ω–Ω—ã–π"
        else:
            return "—Ä–∞–∑–Ω–æ—Å—Ç–æ—Ä–æ–Ω–Ω–∏–π"
    
    def __str__(self):
        return (f"üî∫ –¢—Ä–µ—É–≥–æ–ª—å–Ω–∏–∫ ({self.triangle_type()}): "
                f"{self.side_a}-{self.side_b}-{self.side_c}, —Ü–≤–µ—Ç: {self.color}")

import math

class Vector:
    """–ú–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–∏–π –≤–µ–∫—Ç–æ—Ä —Å –ø–µ—Ä–µ–≥—Ä—É–∂–µ–Ω–Ω—ã–º–∏ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞–º–∏"""

def singleton(cls):
    """–î–µ–∫–æ—Ä–∞—Ç–æ—Ä –¥–ª—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏–∏ –ø–∞—Ç—Ç–µ—Ä–Ω–∞ Singleton"""
    instances = {}
    lock = threading.Lock()
    
    @functools.wraps(cls)
    def get_instance(*args, **kwargs):
        if cls not in instances:
            with lock:
                if cls not in instances:
                    instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    
    return get_instance

class RateLimiter:
    """–î–µ–∫–æ—Ä–∞—Ç–æ—Ä –¥–ª—è –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è —á–∞—Å—Ç–æ—Ç—ã –≤—ã–∑–æ–≤–æ–≤"""
    
    def __init__(self, max_calls=5, time_window=60):
        self.max_calls = max_calls
        self.time_window = time_window
        self.calls = []
        self.lock = threading.Lock()
    
    def __call__(self, func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            with self.lock:
                current_time = time.time()
                
                # –£–¥–∞–ª—è–µ–º —Å—Ç–∞—Ä—ã–µ –≤—ã–∑–æ–≤—ã
                self.calls = [call_time for call_time in self.calls 
                             if current_time - call_time < self.time_window]
                
                # –ü—Ä–æ–≤–µ—Ä—è–µ–º –ª–∏–º–∏—Ç
                if len(self.calls) >= self.max_calls:
                    raise RuntimeError(
                        f"–ü—Ä–µ–≤—ã—à–µ–Ω –ª–∏–º–∏—Ç –≤—ã–∑–æ–≤–æ–≤: {self.max_calls} –≤—ã–∑–æ–≤–æ–≤ "
                        f"–∑–∞ {self.time_window} —Å–µ–∫—É–Ω–¥"
                    )
                
                # –î–æ–±–∞–≤–ª—è–µ–º —Ç–µ–∫—É—â–∏–π –≤—ã–∑–æ–≤
                self.calls.append(current_time)
            
            return func(*args, **kwargs)
        return wrapper

def hash_password(password: str, salt: Optional[str] = None) -> tuple[str, str]:
    """
    –•–µ—à–∏—Ä–æ–≤–∞–Ω–∏–µ –ø–∞—Ä–æ–ª—è —Å —Å–æ–ª—å—é
    
    Args:
        password: –ü–∞—Ä–æ–ª—å –¥–ª—è —Ö–µ—à–∏—Ä–æ–≤–∞–Ω–∏—è
        salt: –°–æ–ª—å (–µ—Å–ª–∏ –Ω–µ —É–∫–∞–∑–∞–Ω–∞, –≥–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç—Å—è –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏)
    
    Returns:
        tuple: (hash, salt)
    """
    if salt is None:
        salt = secrets.token_hex(16)
    
    # –°–æ–∑–¥–∞–µ–º hash —Å —Å–æ–ª—å—é
    password_hash = hashlib.pbkdf2_hmac(
        'sha256',
        password.encode('utf-8'),
        salt.encode('utf-8'),
        100000  # iterations
    )
    
    return password_hash.hex(), salt

    """

    Args:

        bool: True –µ—Å–ª–∏ –ø–∞—Ä–æ–ª—å –≤–µ—Ä–Ω—ã–π
    """
    computed_hash, _ = hash_password(password, salt)
    return computed_hash == password_hash

def format_file_size(size_bytes: int) -> str:
    """
    –§–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ —Ä–∞–∑–º–µ—Ä–∞ —Ñ–∞–π–ª–∞ –≤ —á–µ–ª–æ–≤–µ–∫–æ—á–∏—Ç–∞–µ–º—ã–π –≤–∏–¥

    Examples:

        '1.0 KB'

    if size_bytes == 0:
        return "0 B"
    
    size_names = ["B", "KB", "MB", "GB", "TB"]
    i = 0

    while size >= 1024.0 and i < len(size_names) - 1:
        size /= 1024.0
        i += 1
    
    return f"{size:.1f} {size_names[i]}"

def chunk_list(lst: List[Any], chunk_size: int) -> List[List[Any]]:
    """
    –†–∞–∑–±–∏–µ–Ω–∏–µ —Å–ø–∏—Å–∫–∞ –Ω–∞ —á–∞—Å—Ç–∏ –∑–∞–¥–∞–Ω–Ω–æ–≥–æ —Ä–∞–∑–º–µ—Ä–∞

def deep_merge_dicts(dict1: Dict[str, Any], dict2: Dict[str, Any]) -> Dict[str, Any]:
    """
    –ì–ª—É–±–æ–∫–æ–µ —Å–ª–∏—è–Ω–∏–µ —Å–ª–æ–≤–∞—Ä–µ–π
    
    Args:
        dict1: –ü–µ—Ä–≤—ã–π —Å–ª–æ–≤–∞—Ä—å
        dict2: –í—Ç–æ—Ä–æ–π —Å–ª–æ–≤–∞—Ä—å
    
    Returns:
        Dict: –û–±—ä–µ–¥–∏–Ω–µ–Ω–Ω—ã–π —Å–ª–æ–≤–∞—Ä—å
    """
    result = dict1.copy()
    
    for key, value in dict2.items():
        if (key in result and 
            isinstance(result[key], dict) and 
            isinstance(value, dict)):
            result[key] = deep_merge_dicts(result[key], value)
        else:
            result[key] = value
    
    return result

    """–ö–ª–∞—Å—Å –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è —Ü–µ–ø–æ—á–µ–∫ –≤–∞–ª–∏–¥–∞—Ü–∏–∏"""
    
    def __init__(self, value: Any):
        self.value = value
        self.errors = []
    
    def required(self, message: str = "–ü–æ–ª–µ –æ–±—è–∑–∞—Ç–µ–ª—å–Ω–æ –¥–ª—è –∑–∞–ø–æ–ª–Ω–µ–Ω–∏—è"):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –æ–±—è–∑–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å"""
        if self.value is None or self.value == "":
            self.errors.append(message)
        return self
    
    def min_length(self, length: int, message: str = None):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–π –¥–ª–∏–Ω—ã"""
        if message is None:
            message = f"–ú–∏–Ω–∏–º–∞–ª—å–Ω–∞—è –¥–ª–∏–Ω–∞: {length}"
        
        if isinstance(self.value, str) and len(self.value) < length:
            self.errors.append(message)
        return self
    
    def max_length(self, length: int, message: str = None):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–π –¥–ª–∏–Ω—ã"""
        if message is None:
            message = f"–ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –¥–ª–∏–Ω–∞: {length}"
        
        if isinstance(self.value, str) and len(self.value) > length:
            self.errors.append(message)
        return self
    
    def email(self, message: str = "–ù–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã–π email –∞–¥—Ä–µ—Å"):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ email"""
        if self.value and not is_valid_email(self.value):
            self.errors.append(message)
        return self
    
    def custom(self, validator_func: Callable[[Any], bool], 
              message: str = "–ó–Ω–∞—á–µ–Ω–∏–µ –Ω–µ –ø—Ä–æ—à–ª–æ –≤–∞–ª–∏–¥–∞—Ü–∏—é"):
        """–ü–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å—Å–∫–∞—è –≤–∞–ª–∏–¥–∞—Ü–∏—è"""
        if self.value and not validator_func(self.value):
            self.errors.append(message)
        return self
    
    def validate(self) -> List[str]:
        """–í–æ–∑–≤—Ä–∞—Ç —Å–ø–∏—Å–∫–∞ –æ—à–∏–±–æ–∫"""
        return self.errors
    
    def is_valid(self) -> bool:
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –≤–∞–ª–∏–¥–Ω–æ—Å—Ç–∏"""
        return len(self.errors) == 0
    
    def raise_if_invalid(self):
        """–í—ã–∑–æ–≤ –∏—Å–∫–ª—é—á–µ–Ω–∏—è –ø—Ä–∏ –Ω–∞–ª–∏—á–∏–∏ –æ—à–∏–±–æ–∫"""
        if not self.is_valid():
            raise ValidationError("; ".join(self.errors))

class User:
    """–ú–æ–¥–µ–ª—å –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è —Å–∏—Å—Ç–µ–º—ã"""
    
    # –ö–ª–∞—Å—Å–æ–≤—ã–µ –∞—Ç—Ä–∏–±—É—Ç—ã
    _users_count = 0
    _active_users = set()
    
    def __init__(self, username: str, email: str, password: str, 
                 first_name: str = "", last_name: str = ""):
        """
        –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
        
        Args:
            username: –ò–º—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
            email: Email –∞–¥—Ä–µ—Å
            password: –ü–∞—Ä–æ–ª—å
            first_name: –ò–º—è
            last_name: –§–∞–º–∏–ª–∏—è
        """
        # –í–∞–ª–∏–¥–∞—Ü–∏—è –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        self._validate_user_data(username, email, password)
        
        # –ü—Ä–∏–≤–∞—Ç–Ω—ã–µ –∞—Ç—Ä–∏–±—É—Ç—ã
        self._id = User._users_count + 1
        self._created_at = datetime.now()
        self._password_hash = None
        self._password_salt = None
        self._is_active = True
        self._last_login = None
        
        # –ü—É–±–ª–∏—á–Ω—ã–µ –∞—Ç—Ä–∏–±—É—Ç—ã
        self.username = username
        self.email = email
        self.first_name = first_name
        self.last_name = last_name
        self.roles = []
        self.metadata = {}
        
        # –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –ø–∞—Ä–æ–ª—å
        self.set_password(password)
        
        # –û–±–Ω–æ–≤–ª—è–µ–º —Å—á–µ—Ç—á–∏–∫–∏
        User._users_count += 1
        User._active_users.add(self._id)
    
    @staticmethod
    def _validate_user_data(username: str, email: str, password: str):
        """–í–∞–ª–∏–¥–∞—Ü–∏—è –¥–∞–Ω–Ω—ã—Ö –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è"""
        # –í–∞–ª–∏–¥–∞—Ü–∏—è username
        Validator(username)\
            .required("–ò–º—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –æ–±—è–∑–∞—Ç–µ–ª—å–Ω–æ")\
            .min_length(3, "–ò–º—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å –Ω–µ –º–µ–Ω–µ–µ 3 —Å–∏–º–≤–æ–ª–æ–≤")\
            .max_length(50, "–ò–º—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å –Ω–µ –±–æ–ª–µ–µ 50 —Å–∏–º–≤–æ–ª–æ–≤")\
            .raise_if_invalid()
        
        # –í–∞–ª–∏–¥–∞—Ü–∏—è email
        Validator(email)\
            .required("Email –æ–±—è–∑–∞—Ç–µ–ª–µ–Ω")\
            .email()\
            .raise_if_invalid()
        
        # –í–∞–ª–∏–¥–∞—Ü–∏—è password
        Validator(password)\
            .required("–ü–∞—Ä–æ–ª—å –æ–±—è–∑–∞—Ç–µ–ª–µ–Ω")\
            .min_length(6, "–ü–∞—Ä–æ–ª—å –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –Ω–µ –º–µ–Ω–µ–µ 6 —Å–∏–º–≤–æ–ª–æ–≤")\
            .raise_if_invalid()
    
    @property
    def id(self) -> int:
        """ID –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è (—Ç–æ–ª—å–∫–æ –¥–ª—è —á—Ç–µ–Ω–∏—è)"""
        return self._id
    
    @property
    def created_at(self) -> datetime:
        """–î–∞—Ç–∞ —Å–æ–∑–¥–∞–Ω–∏—è (—Ç–æ–ª—å–∫–æ –¥–ª—è —á—Ç–µ–Ω–∏—è)"""
        return self._created_at
    
    @property
    def is_active(self) -> bool:
        """–°—Ç–∞—Ç—É—Å –∞–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏"""
        return self._is_active
    
    @property
    def full_name(self) -> str:
        """–ü–æ–ª–Ω–æ–µ –∏–º—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è"""
        return f"{self.first_name} {self.last_name}".strip()

    def last_login(self) -> Optional[datetime]:
        """–í—Ä–µ–º—è –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ –≤—Ö–æ–¥–∞"""
        return self._last_login
    
    def set_password(self, password: str):
        """–£—Å—Ç–∞–Ω–æ–≤–∫–∞ –Ω–æ–≤–æ–≥–æ –ø–∞—Ä–æ–ª—è"""
        Validator(password)\
            .required("–ü–∞—Ä–æ–ª—å –æ–±—è–∑–∞—Ç–µ–ª–µ–Ω")\
            .min_length(6, "–ü–∞—Ä–æ–ª—å –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –Ω–µ –º–µ–Ω–µ–µ 6 —Å–∏–º–≤–æ–ª–æ–≤")\
            .raise_if_invalid()
        
        self._password_hash, self._password_salt = hash_password(password)
    
    def check_password(self, password: str) -> bool:
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –ø–∞—Ä–æ–ª—è"""
        if not self._password_hash or not self._password_salt:
            return False
        return verify_password(password, self._password_hash, self._password_salt)
    
    def login(self, password: str) -> bool:
        """–í—Ö–æ–¥ –≤ —Å–∏—Å—Ç–µ–º—É"""
        if not self._is_active:
            raise RuntimeError("–ü–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –¥–µ–∞–∫—Ç–∏–≤–∏—Ä–æ–≤–∞–Ω")
        
        if self.check_password(password):
            self._last_login = datetime.now()
            return True
        return False
    
    def deactivate(self):
        """–î–µ–∞–∫—Ç–∏–≤–∞—Ü–∏—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è"""
        self._is_active = False
        User._active_users.discard(self._id)
    
    def activate(self):
        """–ê–∫—Ç–∏–≤–∞—Ü–∏—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è"""
        self._is_active = True
        User._active_users.add(self._id)
    
    def add_role(self, role: str):
        """–î–æ–±–∞–≤–ª–µ–Ω–∏–µ —Ä–æ–ª–∏"""
        if role not in self.roles:
            self.roles.append(role)
    
    def remove_role(self, role: str):
        """–£–¥–∞–ª–µ–Ω–∏–µ —Ä–æ–ª–∏"""
        if role in self.roles:
            self.roles.remove(role)
    
    def has_role(self, role: str) -> bool:
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞–ª–∏—á–∏—è —Ä–æ–ª–∏"""
        return role in self.roles
    
    def update_metadata(self, **kwargs):
        """–û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö"""
        self.metadata.update(kwargs)
    
    def to_dict(self, include_sensitive: bool = False) -> Dict[str, Any]:
        """–ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –≤ —Å–ª–æ–≤–∞—Ä—å"""
        data = {
            'id': self._id,
            'username': self.username,
            'email': self.email,
            'first_name': self.first_name,
            'last_name': self.last_name,
            'full_name': self.full_name,
            'is_active': self._is_active,
            'created_at': self._created_at.isoformat(),
            'last_login': self._last_login.isoformat() if self._last_login else None,
            'roles': self.roles.copy(),
            'metadata': self.metadata.copy()
        }
        
        if include_sensitive:
            data.update({
                'password_hash': self._password_hash,
                'password_salt': self._password_salt
            })
        
        return data
    
    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> 'User':
        """–°–æ–∑–¥–∞–Ω–∏–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –∏–∑ —Å–ª–æ–≤–∞—Ä—è"""
        # –°–æ–∑–¥–∞–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π –ø–∞—Ä–æ–ª—å (–±—É–¥–µ—Ç –∑–∞–º–µ–Ω–µ–Ω –µ—Å–ª–∏ –µ—Å—Ç—å —Ö–µ—à)
        user = cls(
            username=data['username'],
            email=data['email'],
            password='temporary_password',
            first_name=data.get('first_name', ''),
            last_name=data.get('last_name', '')
        )
        
        # –í–æ—Å—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º —Å–æ—Ö—Ä–∞–Ω–µ–Ω–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ
        if 'password_hash' in data and 'password_salt' in data:
            user._password_hash = data['password_hash']
            user._password_salt = data['password_salt']
        
        user.roles = data.get('roles', [])
        user.metadata = data.get('metadata', {})
        
        if 'is_active' in data:
            if data['is_active']:
                user.activate()
            else:
                user.deactivate()
        
        return user
    
    @classmethod
    def get_user_count(cls) -> int:
        """–ü–æ–ª—É—á–µ–Ω–∏–µ –æ–±—â–µ–≥–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π"""

        return len(cls._active_users)

        return (f"User(id={self._id}, username='{self.username}', "

        if isinstance(other, User):
