# Базовая обработка

    number = int(input("Введите число: "))
    result = 10 / number
    print(f"Результат: {result}")
except ValueError:
    print("Это не число!")
except ZeroDivisionError:
    print("Деление на ноль!")
except Exception as e:
    print(f"Неожиданная ошибка: {e}")

def login_view(request):

        password = request.POST['password']

        if user:
            login(request, user)

### AsyncIO основы
```python
import asyncio
import aiohttp
import time

# Синхронная версия
def fetch_sync(url):
    import requests
    response = requests.get(url)
    return response.json()

def main_sync():
    urls = ['http://api1.com', 'http://api2.com', 'http://api3.com']
    start = time.time()
    results = [fetch_sync(url) for url in urls]
    print(f"Синхронно: {time.time() - start:.2f}с")

# Асинхронная версия
async def fetch_async(session, url):
    async with session.get(url) as response:
        return await response.json()

async def main_async():
    urls = ['http://api1.com', 'http://api2.com', 'http://api3.com']
    start = time.time()
    
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_async(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
    
    print(f"Асинхронно: {time.time() - start:.2f}с")

# Запуск
asyncio.run(main_async())
```

### Кеширование с Redis
```python
import redis
import json
from functools import wraps

# Подключение к Redis
redis_client = redis.Redis(host='localhost', port=6379, db=0)

def cache_result(expiration=300):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # Создаем ключ кеша
            cache_key = f"{func.__name__}:{hash(str(args) + str(kwargs))}"
            
            # Проверяем кеш
            cached = redis_client.get(cache_key)
            if cached:
                return json.loads(cached)
            
            # Выполняем функцию
            result = func(*args, **kwargs)
            
            # Сохраняем в кеш
            redis_client.setex(
                cache_key, 
                expiration, 
                json.dumps(result)
            )
            
            return result
        return wrapper
    return decorator

@cache_result(expiration=600)
def expensive_calculation(n):
    return sum(i**2 for i in range(n))

### ETL с Pandas
```python
import pandas as pd
import numpy as np
from sqlalchemy import create_engine

class DataProcessor:
    def __init__(self, db_url):
        self.engine = create_engine(db_url)
    
    def extract(self, query):
        """Извлечение данных"""
        return pd.read_sql(query, self.engine)
    
    def transform(self, df):
        """Трансформация данных"""
        # Очистка данных
        df = df.dropna()
        
        # Добавление новых колонок
        df['total_amount'] = df['price'] * df['quantity']
        df['profit_margin'] = (df['total_amount'] - df['cost']) / df['total_amount']
        
        # Категоризация
        df['size_category'] = pd.cut(df['quantity'], 
                                   bins=[0, 10, 50, 100, float('inf')],
                                   labels=['Small', 'Medium', 'Large', 'Bulk'])
        
        return df
    
    def load(self, df, table_name):
        """Загрузка данных"""
        df.to_sql(table_name, self.engine, if_exists='replace', index=False)
        
    def run_etl(self):
        """Запуск полного ETL процесса"""
        # Extract
        raw_data = self.extract("SELECT * FROM raw_sales")
        
        # Transform
        processed_data = self.transform(raw_data)
        
        # Load
        self.load(processed_data, 'processed_sales')
        
        print(f"Обработано {len(processed_data)} записей")

### Простое Flask приложение
```python
from flask import Flask, render_template, request

app = Flask(__name__)

@app.route("/")
def index():
    return "<h1>Добро пожаловать!</h1>"

@app.route("/hello/<name>")
def hello(name):
    return f"<h1>Привет, {name}!</h1>"

@app.route("/users", methods=["GET", "POST"])
def users():
    if request.method == "POST":
        # Обработка POST запроса
        name = request.form["name"]
        return f"Создан пользователь: {name}"
    else:
        # Отображение формы
        return render_template("users.html")

if __name__ == "__main__":
    app.run(debug=True)
```

### Обработка форм
```python
from flask import Flask, request, session, flash, redirect

app = Flask(__name__)
app.secret_key = "your-secret-key"

@app.route("/login", methods=["GET", "POST"])
def login():
    if request.method == "POST":
        username = request.form["username"]
        password = request.form["password"]
        
        if username == "admin" and password == "secret":
            session["user"] = username
            flash("Успешный вход!")
            return redirect("/dashboard")
        else:
            flash("Неверный логин или пароль")
    
    return render_template("login.html")

@app.route("/dashboard")
def dashboard():
    if "user" not in session:
        return redirect("/login")
    
    return f"Добро пожаловать, {session['user']}!"
```
